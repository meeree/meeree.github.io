154 bool FileMorphology::GenerateFromHDF(int const N)
155 {
156  // First read in three individual axes, then put into positions,
157  // finally scale and transform into range [-1, 1].
158  std::vector<float> axes[3];
159  static const char* coord_ids[3]{ "/x", "/y", "/z" };
160  int numpnts;
161  try
162  {
163  H5File file(fl_name, H5F_ACC_RDONLY);
164  for (int i = 0; i < 3; ++i)
165  {
166  DataSet axis = file.openDataSet(coord_ids[i]);
167  numpnts = (int)(axis.getStorageSize() / sizeof(float));
168  axes[i].resize(numpnts);
169  axis.read(axes[i].data(), PredType::NATIVE_FLOAT);
170  }
171  }
172  catch (...)
173  {
174  return false;
175  }
176 
177  // Calculate bounding box.
178  glm::vec3 wmin, wmax;
179  for (int i = 0; i < 3; ++i)
180  {
181  wmin[i] = *std::min_element(axes[i].begin(), axes[i].end());
182  wmax[i] = *std::max_element(axes[i].begin(), axes[i].end());
183  }
184 
185  // Check if we have enough points.
186  if (N > numpnts)
187  {
188  std::cerr << "Not enough points in file " << fl_name
189  << ". Have N = " << N << " and " << numpnts
190  << " points available." << std::endl;
191  return false;
192  }
193 
194  // Sample points uniformly so that we have exactly N, not numpnts.
195  int stride{ numpnts / N };
196  for (int i = 0; i < N; ++i)
197  {
198  int const j{ stride * i };
199  positions[i] = { axes[0][j], axes[1][j], axes[2][j] };
200  }
201 
202  // Translate so that min is zero.
203  Translate(-wmin);
204 
205  // Scale so that everything is in the range [0, 2].
206  Scale(glm::vec3(2.0f) / (wmax - wmin));
207 
208  // Finally, translate so that min is -1 and everything is in range [-1, 1].
209  Translate(glm::vec3(-1));
210 
211  return true;
