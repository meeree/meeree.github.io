51  int const num_firing,

55  Float* downstream_copies,

57  Float const* neuron_outputs)

59  __shared__ unsigned int shared_downstream[N_PER_CHUNK];
60  for (int c = 0; c < NCHUNKS; ++c)
61  {
62  // Zero out shared memory
63  for (int tid = threadIdx.x; tid < N_PER_CHUNK; tid += blockDim.x)
64  shared_downstream[tid] = 0;
65 
66  // Get the relevant connectivity and downstream for our chunk.
67  int const* conn = conn_chunks[c];
68  int const* conn_off = conn_off_chunks[c];
69  Float* downstream = downstream_copies + (N_PER_CHUNK * (blockIdx.x + c * gridDim.x));

76  for (; post != conn + conn_off[pre_idx + 1]; ++post)

83  __syncthreads();
